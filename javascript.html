const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); // Load hình ảnh const birdImg = new Image(); birdImg.src = 'https://i.imgur.com/PrIKqEY.png'; const pipeImg = new Image(); pipeImg.src = 'https://i.imgur.com/3w56QSo.png'; const backgroundImg = new Image(); backgroundImg.src = 'https://i.imgur.com/Avt54U3.png'; const gameOverImg = new Image(); gameOverImg.src = 'https://i.imgur.com/qjW3DIh.png'; const startScreenImg = new Image(); startScreenImg.src = 'https://i.imgur.com/VsYPo9N.png'; // Load âm thanh const flapSound = new Audio('https://raw.githubusercontent.com/thaiok2012/Flappy-bird/74b263a7df96e4963f700a3c8b9057e095b42d61/Sound/wing.wav'); // Chim bay const scoreSound = new Audio('https://raw.githubusercontent.com/thaiok2012/Flappy-bird/74b263a7df96e4963f700a3c8b9057e095b42d61/Sound/point.wav'); // Vượt qua cột const gameOverSound = new Audio('https://raw.githubusercontent.com/thaiok2012/Flappy-bird/55b6a990f7daee85ab23d47053a9921c629a2282/Sound/die.wav'); // Chết const smooshSound = new Audio('https://drive.google.com/uc?export=download&id=12beIBx8nh_6Vg7Emf1nzqSlQ8ZOhFRgR'); // Âm thanh môi trường (bạn có thể thay thế nếu muốn) const hitSound = new Audio('https://raw.githubusercontent.com/thaiok2012/Flappy-bird/74b263a7df96e4963f700a3c8b9057e095b42d61/Sound/hit.wav'); // Đâm vào cột let gameOverSoundPlayed = false; // Các biến cơ bản let birdX = 100; let birdY = canvas.height / 2; const birdWidth = 50; const birdHeight = 35; let gravity = 0.5; let velocity = 0; let flapStrength = -10; let pipes = []; let score = 0; let gameSpeed = 2; let pipeGapBase = 200; let pipeGap; let gameStarted = false; let gameOver = false; const pipeWidth = 70; // Hàm vẽ chim function drawBird() { ctx.drawImage(birdImg, birdX - birdWidth / 2, birdY - birdHeight / 2, birdWidth, birdHeight); } // Hàm vẽ ống function drawPipe(pipe) { ctx.save(); ctx.translate(pipe.x + pipeWidth, pipe.topHeight); ctx.rotate(Math.PI); ctx.drawImage(pipeImg, 0, 0, pipeWidth, pipe.topHeight); ctx.restore(); ctx.drawImage(pipeImg, pipe.x, pipe.bottomY, pipeWidth, canvas.height - pipe.bottomY); } // Hàm tạo ống mới function createPipe() { let topHeight = Math.floor(Math.random() * (canvas.height / 2)) + 50; let bottomY = topHeight + pipeGap; pipes.push({ x: canvas.width, topHeight: topHeight, bottomY: bottomY, width: pipeWidth, passed: false }); } // Hàm cập nhật độ khó dựa trên điểm số function updateDifficulty() { if (score >= 7) { gameSpeed = 2 + (score - 6) * 0.1; pipeGap = Math.max(pipeGapBase - (score - 6) * 3, 130); } else { gameSpeed = 2; pipeGap = pipeGapBase; } } // Hàm cập nhật trạng thái trò chơi function update() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Vẽ nền if (backgroundImg.complete) { ctx.drawImage(backgroundImg, 0, 0, canvas.width, canvas.height); } if (gameStarted && !gameOver) { // Cập nhật độ khó updateDifficulty(); // Cập nhật vị trí chim velocity += gravity; birdY += velocity; // Nhảy canvas.addEventListener('mousedown', flap); canvas.addEventListener('touchstart', flap); // Tạo ống mới if (pipes.length === 0 || pipes [pipes.length - 1].x < canvas.width - (300 - score * 5)) { createPipe(); } // Di chuyển ống pipes.forEach(pipe => { pipe.x -= gameSpeed; drawPipe(pipe); // Kiểm tra va chạm if ( birdX + birdWidth / 2 > pipe.x && birdX - birdWidth / 2 < pipe.x + pipeWidth && (birdY - birdHeight / 2 < pipe.topHeight || birdY + birdHeight / 2 > pipe.bottomY) ) { hitSound.play(); // Âm thanh khi chạm vào cột gameOver = true; } // Tăng điểm if (pipe.x + pipeWidth < birdX - birdWidth / 2 && !pipe.passed) { score++; pipe.passed = true; scoreSound.play(); // Âm thanh khi ghi điểm } }); // Va chạm với đất hoặc trời if (birdY + birdHeight / 2 > canvas.height || birdY - birdHeight / 2 < 0) { gameOver = true; } // Vẽ điểm số ctx.fillStyle = 'white'; ctx.font = '30px Arial'; ctx.fillText('Điểm: ' + score, 10, 30); } else if (gameOver) { // Vẽ Game Over if (gameOverImg.complete) { ctx.drawImage(gameOverImg, canvas.width / 2 - gameOverImg.width / 2, canvas.height / 2 - gameOverImg.height / 2); } else { ctx.fillStyle = 'red'; ctx.font = '40px Arial'; ctx.fillText('Game Over!', canvas.width / 2 - 100, canvas.height / 2 - 20); ctx.fillStyle = 'white'; ctx.font = '20px Arial'; ctx.fillText('Điểm số: ' + score, canvas.width / 2 - 70, canvas.height / 2 + 20); ctx.fillText('Nhấn để chơi lại', canvas.width / 2 - 90, canvas.height / 2 + 60); } if (!gameOverSoundPlayed) { gameOverSound.play(); // Âm thanh khi chết gameOverSoundPlayed = true; } canvas.addEventListener('mousedown', resetGame); canvas.addEventListener('touchstart', resetGame); } else { // Vẽ màn hình Bắt đầu if (startScreenImg.complete) { ctx.drawImage(startScreenImg, canvas.width / 2 - startScreenImg.width / 2, canvas.height / 2 - startScreenImg.height / 2); } else { ctx.fillStyle = 'black'; ctx.font = '30px Arial'; ctx.fillText('Nhấn để bắt đầu', canvas.width / 2 - 120, canvas.height / 2); } canvas.addEventListener('mousedown', startGame); canvas.addEventListener('touchstart', startGame); } drawBird(); requestAnimationFrame(update); } function flap() { velocity = flapStrength; flapSound.play(); // Âm thanh khi bay // Bạn có thể thêm smooshSound.play(); ở đây nếu muốn có tiếng gió khi bay } function startGame() { gameStarted = true; gameOver = false; birdY = canvas.height / 2; velocity = 0; pipes = []; score = 0; gameSpeed = 2; pipeGap = pipeGapBase; gameOverSoundPlayed = false; // Tùy chọn: Phát âm thanh gió khi bắt đầu game // smooshSound.loop = true; // smooshSound.play(); canvas.removeEventListener('mousedown', startGame); canvas.removeEventListener('touchstart', startGame); } function resetGame() { gameStarted = true; gameOver = false; birdY = canvas.height / 2; velocity = 0; pipes = []; score = 0; gameSpeed = 2; pipeGap = pipeGapBase; gameOverSoundPlayed = false; // Tùy chọn: Dừng âm thanh gió khi reset game // smooshSound.pause(); // smooshSound.currentTime = 0; canvas.removeEventListener('mousedown', resetGame); canvas.removeEventListener('touchstart', resetGame); } // Khởi tạo khoảng trống ống ban đầu pipeGap = pipeGapBase; // Bắt đầu vòng lặp trò chơi requestAnimationFrame(update);
